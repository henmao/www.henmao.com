## next.js 重定向

```javascript
/** @type {import('next').NextConfig} */
const nextConfig = {
  reactStrictMode: false,
  async redirects() {
    return [
      {
        source: "/",
        destination: "/dashboard",
        permanent: true, // 使用永久重定向（HTTP 301）
      },
    ];
  },
  // output: "standalone",
};

module.exports = nextConfig;
```
